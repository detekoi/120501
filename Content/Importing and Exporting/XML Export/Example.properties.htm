#   The name of the export job to run.

#   The name must match that of an export job defined within esi.manage.

#   To restrict the set of projects to those that have changed since

#   a given date, set the following parameter

#   should by ".xml".

#   Backslashes must be doubled, e.g. c:\\temp\\export.xml

#########################################################################

#             The name of the project sets that the user would like to include in the export

#             Each name must match a project set defined within esi.manage, seperated by a comma.

#   Project Sets must not contain any commas. Leading and trailing spaces for each project set are acceptable.

#             Start Date - Must appear in the format yyyy-mm-dd

#             Interval - Y (Year), Q (Quarter), M (Month), W (Week), D (Day).

#             Duration - Length of time relative to the interval. Must be an integer

Production_Start_Date = 2011-01-01

Production_Interval = M

Production_Duration = 12

#             Last Actuals Date = Must appear in the format yyyy-mm-dd

#             Last Field Estimate Date = Must appear in the format yyyy-mm-dd

Production_LastFieldEstimate_Date = 2010-01-01
